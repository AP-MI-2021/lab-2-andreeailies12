"""
gaseste ultimul nr prim mai mic decat numarul dat
"""

def get_largest_prime_below(n):
    nr = n - 1
    nuprime = 0
    ok = True

    while ok:
        for divizor in range(2, nr):
            if nr%divizor == 0:
                nuprime = nuprime + 1

        if nuprime == 0:
            return nr
        else:
         nr = nr - 1
         nuprime = 0


def test_get_largest_below():
    assert get_largest_prime_below(50) == 47

test_get_largest_below()
